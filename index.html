3.Write the promise function to print numbers from 1 to 7, in which 1 should be printed after 1 sec,
2 should be printed after 2 sec, 3 should be printed after 3 sec, and so on. 

                function count (nums, time){
                    return new promise((res,rej)=>{
                        if(nums>0){
                            setTimeout(()=>{
                                res(nums++)
                            },time);
                        }
                        else{
                            rej("error")
                        }
                    })
                }
            count(0,1000)
            .then((data)=> console.log(data))
            .catch((error)=>console.log(error))
